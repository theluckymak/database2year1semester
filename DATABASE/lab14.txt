 1 Перевести каждую оценку в рейтинговый балл


SELECT *,
    CASE
        WHEN оценка < 3 THEN 0
        WHEN оценка BETWEEN 3 AND 4 THEN 1
        WHEN оценка = 5 THEN 2
    END AS рейтинговый_балл
FROM оценки;

 2 Вывести список оценок и их буквенное обозначение


SELECT оценка,
    CASE
        WHEN оценка = 5 THEN 'отлично'
        WHEN оценка = 4 THEN 'хорошо'
        WHEN оценка = 3 THEN 'удовлетворительно'
        ELSE 'неудовлетворительно'
    END AS буквенное_обозначение
FROM оценки;

 3 Вывести список оценок и указать значение по системе «зачет-незачет»


SELECT оценка,
    CASE
        WHEN оценка >= 4 THEN 'зачет'
        ELSE 'незачет'
    END AS система_зачет_незачет
FROM оценки;

4 Вывести названия групп и названия специальностей

SELECT группы.название_группы,
    CASE
        WHEN группы.название_группы LIKE '%ПИЭ%' THEN 'Прикладная информатика в экономике'
        WHEN группы.название_группы LIKE '%Мат%' THEN 'Математика'
        WHEN группы.название_группы LIKE '%Инф%' THEN 'Информатика'
        ELSE группы.название_группы
    END AS название_специальности
FROM группы;

 5 Вывести фамилии студентов и место прохождения практики


SELECT студенты.фамилия,
    CASE
        WHEN студенты.код_группы = 1 THEN 'Банк УралСиб'
        WHEN студенты.код_группы = 2 THEN 'СберБанк'
        WHEN студенты.код_группы = 3 THEN 'Первомайский'
        WHEN студенты.код_группы = 4 THEN 'РосСельхозБанк'
        ELSE 'Место не указано'
    END AS место_практики
FROM студенты;
